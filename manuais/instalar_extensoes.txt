@echo off
setlocal
title Instalador do Ambiente LaTeX da Tese

REM ============================================================
REM INSTALADOR DO AMBIENTE LATEX DA TESE
REM ESTE ARQUIVO DEVE FICAR DENTRO DA PASTA manual
REM ELE PROCURA tese_ufpi.tex NA PASTA PAI
REM ============================================================

set DELAY=1

REM Pasta onde o .bat esta
set "SCRIPT_DIR=%~dp0"

REM Pasta pai da pasta manual
cd /d "%SCRIPT_DIR%.."
set "ROOT_DIR=%CD%"

REM Volta para a pasta do script
cd /d "%SCRIPT_DIR%"

REM Arquivo principal
set "MAIN_FILE=tese_ufpi.tex"
set "MAIN_PATH=%ROOT_DIR%\%MAIN_FILE%"

echo ====================================================
echo Instalando ambiente para compilar a tese
echo ====================================================
echo.
echo Pasta do script : %SCRIPT_DIR%
echo Pasta do projeto: %ROOT_DIR%
echo Arquivo da tese : %MAIN_PATH%
echo.

if not exist "%MAIN_PATH%" (
    echo ERRO: nao encontrei "%MAIN_FILE%" na pasta que contem a pasta manual.
    echo.
    echo Pressione qualquer tecla para sair.
    pause >nul
    exit /b 1
)

where winget >nul 2>nul
if errorlevel 1 (
    echo ERRO: Winget nao encontrado.
    echo Atualize o Windows ou instale o App Installer.
    echo.
    echo Pressione qualquer tecla para sair.
    pause >nul
    exit /b 1
)

echo [1/6] Instalando Visual Studio Code...
winget install -e --id Microsoft.VisualStudioCode --source winget --accept-source-agreements --accept-package-agreements
timeout /t %DELAY% /nobreak >nul
echo.

echo [2/6] Instalando Git...
winget install -e --id Git.Git --source winget --accept-source-agreements --accept-package-agreements
timeout /t %DELAY% /nobreak >nul
echo.

echo [3/6] Instalando MiKTeX...
winget install -e --id MiKTeX.MiKTeX --source winget --accept-source-agreements --accept-package-agreements
timeout /t %DELAY% /nobreak >nul
echo.

set "CODECMD=%LocalAppData%\Programs\Microsoft VS Code\bin\code.cmd"

if not exist "%CODECMD%" (
    echo AVISO: nao foi possivel localizar o comando code agora.
    echo Abra o VS Code uma vez e execute este arquivo novamente.
    echo.
    echo Pressione qualquer tecla para sair.
    pause >nul
    exit /b 1
)

echo [4/6] Instalando extensao LaTeX Workshop...
call "%CODECMD%" --install-extension James-Yu.latex-workshop
timeout /t %DELAY% /nobreak >nul
echo.

echo [5/6] Instalando extensao GitHub Copilot Chat...
call "%CODECMD%" --install-extension GitHub.copilot-chat
timeout /t %DELAY% /nobreak >nul
echo.

echo [6/6] Instalando GitHub CLI...
winget install -e --id GitHub.cli --source winget --accept-source-agreements --accept-package-agreements
timeout /t %DELAY% /nobreak >nul
echo.

echo Instalando Strawberry Perl...
winget install -e --id StrawberryPerl.StrawberryPerl --source winget --accept-source-agreements --accept-package-agreements
timeout /t 1 /nobreak >nul
echo.
echo Processo concluido.
echo Pressione qualquer tecla para sair.
pause >nul
